object WordCount { def main(args: Array[String]): Unit = { // Check if the file path is provided as an argument if (args.length < 1) { println("Usage: WordCount <file-path>") sys.exit(1) // Exit if no file path is provided } // Get the file path from the command-line arguments val filePath = args(0) // Read the file and count the word occurrences val map = ReadFile.readFile(filePath) // Print the word count results ReadFile.printContent(map) } } object ReadFile { def readFile(filePath: String): Map[String, Int] = { // Open the file and process it line by line val counter = scala.io.Source.fromFile(filePath) .getLines() .flatMap(_.split("\\W+")) // Split by non-word characters (spaces, punctuation) .foldLeft(Map.empty[String, Int]) { (count, word) => // Update the count for each word count + (word -> (count.getOrElse(word, 0) + 1)) } counter // Return the word count map } def printContent(counter: Map[String, Int]): Unit = { // Print each word and its occurrence count for ((word, count) <- counter) { printf("%s : %d\n", word, count) } } }

ENVÍO DE LOS TRABAJOS AL MASTER

Primero creamos un contenedor temporal para poder enviar los trabajos al master
Bash
Copiar
kubectl run -it --rm --image=bitnami/spark:3.5.3 --namespace=spark spark-submit-pod -- bash
​
Luego copiamos el .jar que contiene el trabajo de Scala al contenedor temporal (Abrir un terminal nuevo)
Bash
Copiar
kubectl cp ./target/scala-2.12/wordcount_2.12-1.0.jar spark-submit-pod:/tmp/wordcount_2.12-1.0.jar -n spark
​
Creamos un archivo que contenga las palabras a contar en ./Data/
Bash
Copiar
mkdir Data echo a1 a2 a2 a3 a3 a3 > ./Data/Reading.txt
​
Por último enviamos el trabajo a ejecutar
Bash
Copiar
spark-submit --master spark://spark-master-0.spark-headless.spark.svc.cluster.local:7077 --deploy-mode client --class WordCount --name word-count-job --conf spark.executor.instances=2 --conf spark.executor.cores=1 --conf spark.executor.memory=1G --conf spark.driver.memory=1G /tmp/wordcount_2.12-1.0.jar ./Data/Reading.txt

9- EJECUCIÓN DE UN TRABAJO CON PYSPARK

Primero creamos el .py que contiene el trabajo a ejecutar
Bash
Copiar
notepad WordCount.py
​
Python
Copiar
from pyspark import SparkContext, SparkConf from pyspark.sql import SparkSession import sys def main(file_path): conf = SparkConf().setAppName("WordCount").setMaster("local") sc = SparkContext(conf=conf) text_file = sc.textFile(file_path) word_counts = (text_file.flatMap(lambda line: line.split()).map(lambda word: (word, 1)).reduceByKey(lambda a, b: a + b)) results = word_counts.take(3) for word, count in results: print(f"{word}: {count}") sc.stop() if __name__ == "__main__": main(sys.argv[1])
